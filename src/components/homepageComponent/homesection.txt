  OrbitControls,
  Environment,

  Html,
  useProgress,
} from "@react-three/drei";
import { motion } from "framer-motion";
import { Play } from "lucide-react";

function Loader() {
  const { progress } = useProgress();
  return (
    <Html center>
      <div className="text-white font-semibold text-lg bg-black/70 px-4 py-2 rounded-lg backdrop-blur-sm border border-gray-700">
        {progress.toFixed(0)}% Loaded
      </div>
    </Html>
  );
}

function PrinterModel() {
  const { scene } = useGLTF("/models/printer.glb");
  return (
    <primitive
      object={scene}
      scale={1.5}
      position={[0, -1, 0]}
      rotation={[0, Math.PI / 6, 0]}
    />
  );
}
